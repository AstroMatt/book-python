Add values
----------
* Uses inner join
* ``fill_value``: If data in both corresponding ``Series`` locations is missing the result will be missing

.. code-block:: python

    import numpy as np


    a = pd.Series([1, 1, 1, np.nan], index=['a', 'b', 'c', 'd'])
    # a    1.0
    # b    1.0
    # c    1.0
    # d    NaN
    # dtype: float64

    b = pd.Series([1, np.nan, 1, np.nan], index=['a', 'b', 'x', 'y'])
    # a    1.0
    # b    NaN
    # x    1.0
    # y    NaN
    # dtype: float64

.. code-block:: python

    a = pd.Series([1, 1, 1, np.nan], index=['a', 'b', 'c', 'd'])
    b = pd.Series([1, np.nan, 1, np.nan], index=['a', 'b', 'x', 'y'])

    a + b
    # a    2.0
    # b    NaN
    # c    NaN
    # d    NaN
    # x    NaN
    # y    NaN
    # dtype: float64

.. code-block:: python
    :caption: ``fill_value``: If data in both corresponding ``Series`` locations is missing the result will be missing

    a = pd.Series([1, 1, 1, np.nan], index=['a', 'b', 'c', 'd'])
    b = pd.Series([1, np.nan, 1, np.nan], index=['a', 'b', 'x', 'y'])

    a.add(b, fill_value=0)
    # a    2.0
    # b    1.0
    # c    1.0
    # d    NaN
    # x    1.0
    # y    NaN
    # dtype: float64
